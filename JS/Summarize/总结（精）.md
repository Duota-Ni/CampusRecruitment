### 5.对象（数组）的深克隆和浅克隆  
[对象（数组）的深克隆和浅克隆](../BasicJSExercises/语法/对象（数组）的深克隆和浅克隆.js)
#### 浅克隆  
```js
let obj = {
  a: 100,
  b: [10, 20, 30],
  c: { x: 10 },
  d:/^\d+$/
};
//浅克隆
//多维对象/数组，只把第一层克隆了
//1.
let obj3 = {...obj};
//2.
let obj2 = {};
for(let key in obj){
  if(!obj.hasOwnProperty(key)) break;
  obj2[key] = obj[key]
}
console.log(obj);
console.log(obj2);
console.log(obj === obj2);
obj2.c.x = 1000;
console.log(obj2.c); //{x:1000}
console.log(obj.c); // {x:1000} //这里把二维的原对象的对象里面的属性也给改了（浅克隆）
```
